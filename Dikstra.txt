#include <stdio.h>

#define INF 99999

void dijkstra(int n, int cost[20][20], int source)
{
    int dist[20], visited[20];
    int i, count, u, min;

    // Initialize distance and visited arrays
    for (i = 0; i < n; i++)
    {
        dist[i] = cost[source][i];
        visited[i] = 0;
    }

    dist[source] = 0;
    visited[source] = 1;

    // Main Dijkstra loop
    for (count = 1; count < n; count++)
    {
        min = INF;

        // Find unvisited vertex with minimum distance
        for (i = 0; i < n; i++)
        {
            if (!visited[i] && dist[i] < min)
            {
                min = dist[i];
                u = i;
            }
        }

        visited[u] = 1;

        // Update distances of adjacent vertices
        for (i = 0; i < n; i++)
        {
            if (!visited[i] &&
                dist[u] + cost[u][i] < dist[i])
            {
                dist[i] = dist[u] + cost[u][i];
            }
        }
    }

    printf("\nShortest distances from vertex %d:\n", source);

    for (i = 0; i < n; i++)
    {
        printf("Vertex %d -> %d\n", i, dist[i]);
    }
}

int main()
{
    int n, source;
    int cost[20][20];

    printf("Enter number of vertices: ");
    scanf("%d", &n);

    printf("Enter the cost adjacency matrix:\n");

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            scanf("%d", &cost[i][j]);

            // Replace 0 by INF except diagonal
            if (i != j && cost[i][j] == 0)
                cost[i][j] = INF;
        }
    }

    printf("Enter source vertex: ");
    scanf("%d", &source);

    dijkstra(n, cost, source);

    return 0;
}
